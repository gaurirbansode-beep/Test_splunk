# Extracted Splunk logger functions

def initialize_service():
    return client.connect(host="localhost", port=8089, username="admin", password="changeme")

def build_query(index, condition):
    return f"search index={index} {condition}"

def submit_event(message):
    svc = initialize_service()
    svc.indexes["main"].submit(message)

def persist_event(event):
    submit_event(json.dumps(event))

def execute_search(query):
    svc = initialize_service()
    return svc.jobs.create(query)

def search_error_events():
    q = "search index=main ERROR | stats count"
    return execute_search(q)

def search_by_host(host):
    q = f"search index=main host={host} | stats count"
    return execute_search(q)

def search_with_time():
    q = "search index=main | timechart count"
    return execute_search(q)

def handle_event(user, action):
    evt = create_event(user, action)
    persist_event(evt)

def conditional_eval(value):
    q = f"search index=main | eval status=if(count>{value},\"high\",\"low\")"
    return execute_search(q)

def search_with_lookup():
    q = "search index=main | lookup user_lookup user OUTPUT role"
    return execute_search(q)

def append_search():
    q = "search index=main | append [ search index=backup ]"
    return execute_search(q)

def transaction_search():
    q = "search index=main | transaction user"
    return execute_search(q)

def dedup_hosts():
    q = "search index=main | dedup host"
    return execute_search(q)

def sort_results():
    q = "search index=main | sort - count"
    return execute_search(q)

def rare_hosts():
    q = "search index=main | rare host"
    return execute_search(q)

def top_hosts():
    q = "search index=main | top host"
    return execute_search(q)

def table_view():
    q = "search index=main | table host, source"
    return execute_search(q)

def rex_extract():
    q = "search index=main | rex \"user=(?<user>\\w+)\""
    return execute_search(q)

def build_complex_query(index, field):
    return f"search index={index} | stats count by {field}"

def execute_complex(index, field):
    return execute_search(build_complex_query(index, field))

def search_custom(condition):
    q = build_query("main", condition)
    return execute_search(q)

def aggregate_by_source():

    return execute_search(q)

def aggregate_by_type():

def search_metrics():

    return execute_search(q)

def build_error_query():
    return "search index=main ERROR | table host, source"

def execute_error_table():
    return execute_search(build_error_query())

def search_last_hour():
    q = "search index=main earliest=-1h | stats count"
    return execute_search(q)

def search_today():
    q = "search index=main earliest=@d | stats count"
    return execute_search(q)
