# =========================================================
# ---------------- NON-SPLUNK FUNCTIONS (1–30)
# =========================================================

def add(a, b):
    return a + b

def subtract(a, b):
    return a - b

def multiply(a, b):
    return a * b

def divide(a, b):
    return a / b if b != 0 else None

def greet(name):
    return f"Hello {name}"

def calculate_square(n):
    return n * n

def calculate_cube(n):
    return n ** 3

def reverse_string(text):
    return text[::-1]

def fetch_data_from_api(url):
    response = requests.get(url)
    return response.json()

def parse_json(data):
    return json.loads(data)

def log_info(message):
    logger = logging.getLogger("app")
    logger.info(message)

def log_warning(message):
    logger = logging.getLogger("app")
    logger.warning(message)

def connect_database(conn):
    return conn.cursor()

def fetch_users(conn):
    cursor = conn.cursor()
    cursor.execute("SELECT * FROM users")
    return cursor.fetchall()

def save_to_file(filename, content):
    with open(filename, "w") as f:
        f.write(content)

def read_file(filename):
    with open(filename, "r") as f:
        return f.read()

def calculate_average(numbers):
    return sum(numbers) / len(numbers)

def find_max(numbers):
    return max(numbers)

def find_min(numbers):
    return min(numbers)

def sort_numbers(numbers):
    return sorted(numbers)

def filter_even(numbers):
    return [n for n in numbers if n % 2 == 0]

def filter_odd(numbers):
    return [n for n in numbers if n % 2 != 0]

def uppercase_text(text):
    return text.upper()

def lowercase_text(text):
    return text.lower()

def capitalize_text(text):
    return text.capitalize()

def count_words(text):
    return len(text.split())

def merge_dicts(d1, d2):
    return {**d1, **d2}

def check_prime(n):
    if n <= 1:
        return False
    for i in range(2, n):
        if n % i == 0:
            return False
    return True

def generate_range(n):
    return list(range(n))

def print_message(msg):
    print(msg)

def splunk_connect():
    return client.connect(host="localhost", port=8089,
                          username="admin", password="changeme")

def splunk_log_event(event):
    service = splunk_connect()
    service.indexes["main"].submit(event)

def splunk_log_error(error):
